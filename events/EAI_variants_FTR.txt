#FTR I

country_event = { #475 XP
	id = EAI_V_FTR.1
	hidden = yes
	mean_time_to_happen = { days = 10 }
	trigger = {
		is_ai = yes
		has_tech = fighter1 NOT = { has_tech = fighter2 is_researching_technology = fighter2 }
		NOT = { has_country_flag = EAI_VARIANT/ftr_I_4/4 }
		
		OR = { is_major = no has_air_experience > 450 } #majors dont bother upgrading these

		#XP REQ
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/ftr_I_1/4 } } has_air_experience > 130 }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/ftr_I_2/4 } has_country_flag = EAI_VARIANT/ftr_I_1/4 } has_air_experience > 150 }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/ftr_I_3/4 } has_country_flag = EAI_VARIANT/ftr_I_2/4 } has_air_experience > 195 }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/ftr_I_4/4 } has_country_flag = EAI_VARIANT/ftr_I_3/4 } has_air_experience > 425 }
	}

	immediate = {
		log = "[GetYear] [GetMonth] | AI | [Root.GetName] | VARIANTS: upgrading fighter1"

		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/ftr_I_1/4 }										     has_air_experience > 130 } set_country_flag = EAI_VARIANT/ftr_I_1/4 air_experience = -130 create_equipment_variant = { type = fighter_equipment_1 upgrades = { plane_range_upgrade = 0 plane_gun_upgrade = 2 plane_engine_upgrade = 2 plane_reliability_upgrade = 0 } } }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/ftr_I_2/4 } has_country_flag = EAI_VARIANT/ftr_I_1/4 has_air_experience > 150 } set_country_flag = EAI_VARIANT/ftr_I_2/4 air_experience = -150 create_equipment_variant = { type = fighter_equipment_1 upgrades = { plane_range_upgrade = 0 plane_gun_upgrade = 2 plane_engine_upgrade = 3 plane_reliability_upgrade = 2 } } }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/ftr_I_3/4 } has_country_flag = EAI_VARIANT/ftr_I_2/4 has_air_experience > 195 } set_country_flag = EAI_VARIANT/ftr_I_3/4 air_experience = -195 create_equipment_variant = { type = fighter_equipment_1 upgrades = { plane_range_upgrade = 0 plane_gun_upgrade = 4 plane_engine_upgrade = 4 plane_reliability_upgrade = 2 } } }
