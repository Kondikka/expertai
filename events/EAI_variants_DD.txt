#DD II

country_event = { #175 XP
	id = EAI_V_DD.2
	hidden = yes
	mean_time_to_happen = { days = 10 }
	trigger = {
		is_ai = yes
		has_tech = basic_destroyer NOT = { has_tech = improved_destroyer is_researching_technology = improved_destroyer }
		NOT = { has_country_flag = EAI_VARIANT/dd_II_4/4 }
		
		#FOCUS
		has_country_flag = EAI_FOCUS/dd_navy_focus_1

		#XP REQ
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_1/4 } } has_navy_experience > 175 }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_2/4 } has_country_flag = EAI_VARIANT/dd_II_1/4 } has_navy_experience > 230 }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_3/4 } has_country_flag = EAI_VARIANT/dd_II_2/4 } has_navy_experience > 450 }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_4/4 } has_country_flag = EAI_VARIANT/dd_II_3/4 } has_navy_experience > 450 }

		#PRIORITIES
		if = { limit = { has_country_flag = EAI_FOCUS/cv_navy_focus_1 }
			EAI_carrier_developed = yes
		}
	}

	immediate = {
		log = "[GetYear] [GetMonth] | AI | [Root.GetName] | VARIANTS: upgrading basic_destroyer"

		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_1/4 }											 has_navy_experience > 175 } set_country_flag = EAI_VARIANT/dd_II_1/4 navy_experience = -175 create_equipment_variant = { type = destroyer_2 upgrades = { destroyer_engine_upgrade = 0 ship_anti_air_upgrade = 0 ship_torpedo_upgrade = 5 ship_ASW_upgrade = 0 } } }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_2/4 } has_country_flag = EAI_VARIANT/dd_II_1/4 has_navy_experience > 230 } set_country_flag = EAI_VARIANT/dd_II_2/4 navy_experience = -230 create_equipment_variant = { type = destroyer_2 upgrades = { destroyer_engine_upgrade = 4 ship_anti_air_upgrade = 0 ship_torpedo_upgrade = 5 ship_ASW_upgrade = 0 } } }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_3/4 } has_country_flag = EAI_VARIANT/dd_II_2/4 has_navy_experience > 225 } set_country_flag = EAI_VARIANT/dd_II_3/4 navy_experience = -225 create_equipment_variant = { type = destroyer_2 upgrades = { destroyer_engine_upgrade = 5 ship_anti_air_upgrade = 0 ship_torpedo_upgrade = 5 ship_ASW_upgrade = 2 } } }
		if = { limit = { NOT = { has_country_flag = EAI_VARIANT/dd_II_4/4 } has_country_flag = EAI_VARIANT/dd_II_3/4 has_navy_experience > 270 } set_country_flag = EAI_VARIANT/dd_II_4/4 navy_experience = -270 create_equipment_variant = { type = destroyer_2 upgrades = { destroyer_engine_upgrade = 5 ship_anti_air_upgrade = 0 ship_torpedo_upgrade = 5 ship_ASW_upgrade = 5 } } }
	}

	option = { }
}
