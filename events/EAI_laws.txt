country_event = { # STORE PP FOR LAW CHANGES
	id = EAI_L.0
	hidden = yes

	mean_time_to_happen = { days = 2 }

	trigger = {
		is_ai = yes

		has_political_power > 50

		check_variable = { EAI_stored_PP < 150 }
	}

	option = {
		add_political_power = -50

		if = { limit = { NOT = { has_variable = EAI_stored_PP } }

			set_variable = { EAI_stored_PP = 50 }

			else = { add_to_variable = { EAI_stored_PP = 50 } }
		}
	}
}

#PARTIAL MOBILIZATION
country_event = {
	id = EAI_L.1
	hidden = yes

	mean_time_to_happen = { days = 2 }

	trigger = {
		is_ai = yes

		OR = {
			has_political_power > 130

			AND = { NOT = { check_variable = { EAI_stored_PP < 150 } } NOT = { EAI_prioritise_conscription_law = yes } }
		}

		################### requirements

		EAI_can_take_partial_economic_mobilisation = yes

		################### law priorities

		OR = {
			AND = {
				has_war = yes

				NOT = { EAI_can_upgrade_manpower_law = yes }
			}
			AND = {
				has_war = no
			}
		}
	}

	option = {
		if = { limit = { NOT = { check_variable = { EAI_stored_PP < 150 } } NOT = { EAI_prioritise_conscription_law = yes } }

			add_to_variable = { EAI_stored_PP = -150 }

			else = { add_political_power = -150 }
		}

		add_ideas = partial_economic_mobilisation
	}
}
