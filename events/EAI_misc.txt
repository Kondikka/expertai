country_event = {
	id = EAI_M.1001
	title = EAI_M.1001.t
	desc = EAI_M.1001.d

	picture = GFX_report_event_expert_ai

	is_triggered_only = yes

	immediate = { #set default historical settings
		if = { limit = { is_historical_focus_on = yes NOT = { has_global_flag = EAI_normal } }
			set_global_flag = EAI_normal

			if = { limit = { JAP = { is_ai = yes } }
				JAP = { add_ideas = neutrality_idea3 }
				set_global_flag = EAI_neutral_jap
			}

			if = { limit = { SPR = { is_ai = yes } }
				set_global_flag = EAI_neutral_spr
			}
		}
	}

	#################################################

	option = {
		name = EAI_M.1001.o1a # "Historical off"
		trigger = { NOT = { has_global_flag = EAI_general_historical } }
		hidden_effect = {
			set_global_flag = EAI_general_historical

			country_event = EAI_M.1002
		}
		
		custom_effect_tooltip = EAI_M.1001.o1a.tt
	}
	option = {
		name = EAI_M.1001.o1b # "Historical on"
		trigger = { has_global_flag = EAI_general_historical }
		hidden_effect = {
			clr_global_flag = EAI_general_historical

			country_event = EAI_M.1002
		}
		
		custom_effect_tooltip = EAI_M.1001.o1b.tt
	}

	#################################################

	option = {
		name = EAI_M.1001.o2d # "Insane"
		trigger = { has_global_flag = EAI_general_difficulty_insane }
		hidden_effect = {
			clr_global_flag = EAI_general_difficulty_insane
			set_global_flag = EAI_general_difficulty_easy

			country_event = EAI_M.1002
		}

		custom_effect_tooltip = EAI_M.1001.o2d.tt
	}
	option = {
		name = EAI_M.1001.o2c # "Very Hard"
		trigger = { has_global_flag = EAI_general_difficulty_very_hard }

			clr_global_flag = EAI_general_difficulty_very_hard

	option = {

			country_event = EAI_M.1002
		}
		
		custom_effect_tooltip = EAI_M.1001.o2b.tt
	}
	option = {
		name = EAI_M.1001.o2a # "Normal"
		trigger = {
			NOT = {
				has_global_flag = EAI_general_difficulty_easy
				has_global_flag = EAI_general_difficulty_hard
				has_global_flag = EAI_general_difficulty_very_hard
				has_global_flag = EAI_general_difficulty_insane
			}
		}
		hidden_effect = {
			set_global_flag = EAI_general_difficulty_hard

			country_event = EAI_M.1002
		}
		
		custom_effect_tooltip = EAI_M.1001.o2a.tt
	}
	option = {
		name = EAI_M.1001.o2e # "Easy"
		trigger = { has_global_flag = EAI_general_difficulty_easy }
		hidden_effect = {
			clr_global_flag = EAI_general_difficulty_easy

			country_event = EAI_M.1002
		}
		
		custom_effect_tooltip = EAI_M.1001.o2e.tt
	}

	#################################################

	option = {
		name = EAI_M.1001.o3 # "Customise"
		hidden_effect = {
			if = { limit = { has_global_flag = EAI_general_historical }
				EAI_set_historical = yes
			}

			if = { limit = { has_global_flag = EAI_general_difficulty_insane }
				set_global_flag = EAI_global_desperate_defense
				set_global_flag = EAI_very_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_1
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_very_hard }
				set_global_flag = EAI_global_desperate_defense
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_2
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_hard }
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_3
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_easy }
				clr_global_flag = EAI_lend_lease
				set_global_flag = EAI_no_resistance_reduction

				every_country = { limit = { OR = { has_idea = ns has_idea = ns2 } }
					remove_ideas = ns
					remove_ideas = ns2
				}
			}
		
			country_event = EAI_M.1000
		}
	}
	option = {
		name = EAI_M.1001.o4 # "Exit"
		hidden_effect = {
			if = { limit = { has_global_flag = EAI_general_historical }
				EAI_set_historical = yes
			}

			if = { limit = { has_global_flag = EAI_general_difficulty_insane }
				set_global_flag = EAI_global_desperate_defense
				set_global_flag = EAI_very_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_1
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_very_hard }
				set_global_flag = EAI_global_desperate_defense
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_2
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_hard }
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_3
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_easy }
				clr_global_flag = EAI_lend_lease
				set_global_flag = EAI_no_resistance_reduction

				every_country = { limit = { OR = { has_idea = ns has_idea = ns2 } }
					remove_ideas = ns
					remove_ideas = ns2
				}
			}

			every_country = {
				limit = {
					is_ai = yes
					NOT = { has_country_flag = EAI_army_config }
				}
				country_event = EAI_CFG.0
			}
		}
	}
}

		NOT = { major_country = yes }

			AND = { original_research_slots > 2 original_research_slots < 4 NOT = { has_country_flag = slots_reduced_2 } }
