		if = { limit = { is_historical_focus_on = yes NOT = { has_global_flag = EAI_normal } }

		name = EAI_M.1001.o1b # "Mode: Historical"

	}

	#################################################

	option = {
		name = EAI_M.1001.o3 # "Customise"
		hidden_effect = {
			if = { limit = { has_global_flag = EAI_general_historical }
				EAI_set_historical = yes
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_insane }
				set_global_flag = EAI_global_desperate_defense
				set_global_flag = EAI_very_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_1
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_very_hard }
				set_global_flag = EAI_global_desperate_defense
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_2
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_hard }
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_3
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_easy }
				clr_global_flag = EAI_lend_lease
				set_global_flag = EAI_no_resistance_reduction

				every_country = { limit = { OR = { has_idea = ns has_idea = ns2 } }
					remove_ideas = ns
					remove_ideas = ns2
				}
			}

			country_event = EAI_M.1000
		}
	}
	option = {
		name = EAI_M.1001.o4 # "Exit"
		hidden_effect = {
			if = { limit = { has_global_flag = EAI_general_historical }
				EAI_set_historical = yes
			}

			if = { limit = { has_global_flag = EAI_general_difficulty_insane }

				set_global_flag = EAI_attrition_1
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_very_hard }
				set_global_flag = EAI_global_desperate_defense
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_2
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_hard }
				set_global_flag = EAI_hard_enemies
				set_global_flag = EAI_attrition
				set_global_flag = EAI_attrition_3
			}
			if = { limit = { has_global_flag = EAI_general_difficulty_easy }
				clr_global_flag = EAI_lend_lease
				set_global_flag = EAI_no_resistance_reduction

				every_country = { limit = { OR = { has_idea = ns has_idea = ns2 } }
					remove_ideas = ns
					remove_ideas = ns2
				}
			}

			every_country = {
				limit = {
					is_ai = yes
					NOT = { has_country_flag = EAI_army_config }
				}
				country_event = EAI_CFG.0
			}
		}
	}
}
