# checks the current variant status for a tech when sorting out upgrade priorities between various techs
EAI_marm_developed = {
	OR = {
		AND = { has_country_flag = EAI_VARIANT/marm_I_2/4 has_tech = basic_medium_tank NOT = { has_tech = improved_medium_tank is_researching_technology = improved_medium_tank } }
		AND = { has_country_flag = EAI_VARIANT/marm_II_2/4 has_tech = improved_medium_tank NOT = { has_tech = advanced_medium_tank is_researching_technology = advanced_medium_tank } }
		AND = { has_country_flag = EAI_VARIANT/marm_III_2/4 has_tech = advanced_medium_tank NOT = { has_tech = main_battle_tank is_researching_technology = main_battle_tank } }
		AND = { has_country_flag = EAI_VARIANT/marm_IV_2/4 has_tech = main_battle_tank }
	}
}
