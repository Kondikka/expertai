any_willing_to_guarantee_nation = {
	is_ai = yes

	NOT = { tag = FROM tag = ROOT }

	OR = { #which ideologies this country is willing to defend
		AND = {
			has_government = democratic
			FROM = { OR = { has_government = democratic has_government = neutrality } }
		}
		AND = {
			has_government = communism
			FROM = { OR = { has_government = communism has_government = neutrality } }
		}
		AND = {
			has_government = fascism
			FROM = { OR = { has_government = fascism has_government = neutrality } }
		}
		AND = {
			has_government = neutrality
		}
	}

	OR = { #ignore if the invader has the same ideology as this AND target is not the same ideology
		AND = {
			has_government = democratic
			OR = {
				NOT = { ROOT = { has_government = democratic } }
				AND = {
					ROOT = { has_government = democratic }
					OR = {
						FROM = { has_government = democratic }
						FROM = { has_government = neutrality }
					}
				}
			}
		}

		AND = {
			has_government = communism
			OR = {
				NOT = { ROOT = { has_government = communism } }
				AND = {
					ROOT = { has_government = communism }
					FROM = { has_government = communism }
				}
			}
		}

		AND = {
			has_government = fascism
			OR = {
				NOT = { ROOT = { has_government = fascism } }
				AND = {
					ROOT = { has_government = fascism }
					FROM = { has_government = fascism }
				}
			}
		}

		AND = {
			has_government = neutrality
		}
	}

	NOT = { #reasons NOT to oppose the invader
		neutral_country = yes
		has_war = yes
		is_in_faction_with = ROOT
		has_non_aggression_pact_with = ROOT
		has_opinion = { target = ROOT value > 50 }
		strength_ratio = { tag = ROOT ratio < 0.2 }
		FROM = { strength_ratio = { tag = ROOT ratio > 2 } }
	}

	OR = { #location
		AND = { is_faction_leader = yes tag = ENG } #if a faction leader is willing to defend globally
		
		AND = { #original capitals must be connected through land
			any_state = {
				is_controlled_by = PREV
				is_in_home_area = yes
				any_neighbor_state = {
					is_controlled_by = FROM
					is_in_home_area = yes
				}
			}
			OR = {
				AND = { capital_scope = { is_on_continent = europe } FROM = { has_country_flag = origin_europe } }
				AND = { capital_scope = { is_on_continent = north_america } FROM = { has_country_flag = origin_north_america } }
				AND = { capital_scope = { is_on_continent = south_america } FROM = { has_country_flag = origin_south_america } }
				AND = { capital_scope = { is_on_continent = africa } FROM = { has_country_flag = origin_africa } }
				AND = { capital_scope = { is_on_continent = asia } FROM = { has_country_flag = origin_asia } }
				AND = { capital_scope = { is_on_continent = middle_east } FROM = { has_country_flag = origin_middle_east } }
			}
		}
		AND = { #only if the capitals are connected to the overlord
			any_state = {
				is_controlled_by = PREV
				is_in_home_area = yes
				any_neighbor_state = {
					is_controlled_by = FROM
					is_in_home_area = yes
				}
			}
			OR = {
				AND = { capital_scope = { is_on_continent = europe } FROM = { has_country_flag = origin_europe } }
				AND = { capital_scope = { is_on_continent = north_america } FROM = { has_country_flag = origin_north_america } }
				AND = { capital_scope = { is_on_continent = south_america } FROM = { has_country_flag = origin_south_america } }
				AND = { capital_scope = { is_on_continent = africa } FROM = { has_country_flag = origin_africa } }
				AND = { capital_scope = { is_on_continent = asia } FROM = { has_country_flag = origin_asia } }
				AND = { capital_scope = { is_on_continent = middle_east } FROM = { has_country_flag = origin_middle_east } }
			}
		}
		AND = { #subjects
			any_country = {
				is_subject_of = PREV
				any_state = {
					is_controlled_by = PREV
					is_in_home_area = yes
					any_neighbor_state = {
						is_controlled_by = FROM
						is_in_home_area = yes
					}
				}
				OR = {
					AND = { capital_scope = { is_on_continent = europe } FROM = { has_country_flag = origin_europe } }
					AND = { capital_scope = { is_on_continent = north_america } FROM = { has_country_flag = origin_north_america } }
					AND = { capital_scope = { is_on_continent = south_america } FROM = { has_country_flag = origin_south_america } }
					AND = { capital_scope = { is_on_continent = africa } FROM = { has_country_flag = origin_africa } }
					AND = { capital_scope = { is_on_continent = asia } FROM = { has_country_flag = origin_asia } }
					AND = { capital_scope = { is_on_continent = middle_east } FROM = { has_country_flag = origin_middle_east } }
				}
			}
		}
	}

eligible_for_LL_receiver = {
	NOT = { has_war_with = ROOT }

	########################################

	#SPECIAL RULES

	#axis shouldn't lend lease japan
	if = { limit = { ROOT = { is_in_faction_with = GER has_war = yes } OR = { original_tag = JAP is_puppet_of = JAP } }
		is_in_faction_with = GER
	}

	if = { limit = { ROOT = { OR = { original_tag = JAP is_puppet_of = JAP } has_war = yes } is_in_faction_with = GER }
		ROOT = { is_in_faction_with = GER }
	}

	########################################

	OR = { #cross ideology LL logic
		AND = {
			OR = {
				has_government = democratic
				AND = { has_offensive_war = no has_government = communism }
				AND = { has_offensive_war = no has_government = neutrality }
				any_enemy_country = { has_war_with = ROOT } #enemy of my enemy is my friend
			}
			ROOT = { has_government = democratic }
		}
		AND = {
			OR = {
				has_government = communism
				AND = { has_offensive_war = no has_government = neutrality }
			}
			ROOT = { has_government = communism }
		}
		AND = {
			OR = {
				has_government = fascism
				AND = { has_offensive_war = no has_government = neutrality }
			}
			ROOT = { has_government = fascism }
		}
		AND = {
			OR = {
				AND = { has_offensive_war = no has_government = democratic }
				AND = { has_offensive_war = no has_government = communism }
				AND = { has_offensive_war = no has_government = neutrality }
				any_enemy_country = { has_war_with = ROOT } #enemy of my enemy is my friend
			}
			ROOT = { has_government = neutrality }
		}
		is_in_faction_with = ROOT #bypass for faction members in case of unholy alliances
	}

	OR = { #why should it send LL?
		any_enemy_country = { has_war_with = ROOT } #enemy of my enemy is my friend
		ROOT = { has_opinion = { target = PREV value > 0 } } #friendly countries
		is_in_faction_with = ROOT #faction members

		AND = { any_enemy_country = { has_government = fascism } ROOT = { NOT = { has_government = fascism } } } #support wars against other ideologies
		AND = { any_enemy_country = { has_government = communism } ROOT = { NOT = { has_government = communism } } }
		AND = { any_enemy_country = { has_government = democratic } ROOT = { NOT = { has_government = democratic } } }

		AND = { has_government = fascism ROOT = { has_government = fascism } }
		AND = { has_government = democratic ROOT = { has_government = democratic } }
	}
