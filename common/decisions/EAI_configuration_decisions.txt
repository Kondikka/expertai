EAI_config_target_modified_DECISIONS = {
	
	EAI_config_target_modified_DECISION = {
		allowed = {}

		available = {
			FROM = {
				NOT = { has_country_flag = EAI_AI_is_configuring_this_country has_country_flag = EAI_player_is_configuring_this_country }
			}
			hidden_trigger = {
				NOT = {
					has_country_flag = EAI_player_has_AI_army_config_open
					has_country_flag = EAI_player_has_AI_army_config_open_DIPLOMATIC
				}
			}
		}

		visible = {
			has_country_flag = EAI_show_AI_config_decisions
			FROM = {
				exists = yes
				OR = {
					has_country_flag = EAI_country_bonus
					has_country_flag = EAI_desperate_defense
					has_country_flag = EAI_division_spawning
				}
			}
		}

		target_trigger = {
			is_ai = no
			FROM = {
				is_ai = yes
			}
		}

		cost = 0

		days_re_enable = -1

		complete_effect = {
			hidden_effect = {
				set_country_flag = EAI_player_has_AI_army_config_open

				FROM = { save_event_target_as = EAI_AI_customisation set_country_flag = EAI_player_is_configuring_this_country }

				set_variable = { ROOT.EAI_dd_unit_count = FROM.EAI_dd_unit_count }

				country_event = EAI.10
			}
		}

		ai_will_do = {
			factor = 0
		}
	}
}
