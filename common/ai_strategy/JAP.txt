EAI_c_tib = {
	enable = {
		has_country_flag = EAI_JAP_invade_TIB
		has_government = fascism
		NOT = { 
			JAP = { has_capitulated = yes }
			JAP = { exists = no }
			JAP = { is_subject = yes }
		}
		NOT = { 
			TIB = { has_capitulated = yes }
			TIB = { exists = no }
			TIB = { is_subject = yes }
			TIB = { has_government = fascism }
			is_in_faction_with = TIB
			has_war_with = TIB
		}
	}
	abort = {
		NOT = {
			AND = {
				has_country_flag = EAI_JAP_invade_TIB
				has_government = fascism
				NOT = { 
					JAP = { has_capitulated = yes }
					JAP = { exists = no }
					JAP = { is_subject = yes }
				}
				NOT = { 
					TIB = { has_capitulated = yes }
					TIB = { exists = no }
					TIB = { is_subject = yes }
					TIB = { has_government = fascism }
					is_in_faction_with = TIB
					has_war_with = TIB
				}
			}
		}
	}
	
	ai_strategy = {
		type = antagonize
		id = "TIB"
		value = 500
	}
	ai_strategy = {
		type = conquer
		id = "TIB"
		value = 500
	}
	ai_strategy = {
		type = ignore
		id = "TIB"
		value = -500
	}
}
