	}
	
	abort = {
		OR = {
			NOT = { SOV = { has_war_with = GER } GER = { is_justifying_wargoal_against = SOV focus_progress = { focus = GER_ussr_war_goal progress > 0 } } }
			MEN = { is_in_faction_with = GER }
			has_war_with = SOV has_war_with = MEN

			is_justifying_wargoal_against = SOV is_justifying_wargoal_against = MEN
			has_wargoal_against = SOV has_wargoal_against = MEN
