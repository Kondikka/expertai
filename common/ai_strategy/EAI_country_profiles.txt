GER_strategy_1 = { #increase prewar armor production
	enable = {
		original_tag = GER
		is_historical_focus_on = yes
		
		OR = {
			ai_has_role_template = armor
			ai_has_role_template = heavy_armor
		}
	}
	
	ai_strategy = { type = equipment_production_factor id = armor               value = 25 }
	
	ai_strategy = { type = equipment_production_factor id = infantry            value = -10 }
	ai_strategy = { type = equipment_production_factor id = artillery           value = -10 }
}

GER_strategy_2 = { #light tanks are low prio
	enable = {
		original_tag = GER
		is_historical_focus_on = yes

		OR = {
			AND = {
				has_country_flag = marm_army_focus
				NOT = { has_country_flag = marm_template }
			}
			AND = {
				has_country_flag = harm_army_focus
				NOT = { has_country_flag = harm_template }
			}
		}
	}
	abort = {
		OR = {
			AND = {
				has_country_flag = marm_army_focus
				has_country_flag = marm_template
			}
			AND = {
				has_country_flag = harm_army_focus
				has_country_flag = harm_template
			}
		}
	}

	ai_strategy = { type = equipment_production_factor id = armor               value = -200 }
}

	enable = {
		original_tag = JAP

		OR = {
			has_war_with = CHI
			date < 1939.1.1
		}

		NOT = { any_enemy_country = { has_deployed_air_force_size = { size > 500 } } }
	}
	abort = {
		OR = {
			AND = {
				NOT = { has_war_with = CHI }
				NOT = { date < 1939.1.1 }
			}
			any_enemy_country = { has_deployed_air_force_size = { size > 500 } }
		}
	}

	ai_strategy = { type = air_factory_balance                                  value = -50 }
}
