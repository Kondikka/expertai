#	tactical_bomber = tactical bomber
#	interceptor = naval bomber
#	naval_bomber = cv fighter + cv naval bomber
#	strategic_bomber = strategic_bomber
#	air_transport = disabled (transport, h. fighter, everything else)

#	capital_ship = battleship,  heavy battleship
#	screen_ship = destroyer, light cruiser
#	carrier = carrier
#	submarine = disabled (subs, BC, CA)

str_aircraft_production = {
	enable = {
		#conditions for building this
		has_tech = strategic_bomber1
		
		if = { limit = { has_country_flag = str_air_focus_1 }
			num_of_military_factories > 60
			
			else = {
				num_of_military_factories > 249
				date > 1944.1.1
				OR = { alliance_strength_ratio > 1.0 NOT = { any_neighbor_country = { has_war_with = ROOT } } }
			}
		}

		#prioritise building fighters
		if = { limit = { enemies_strength_ratio > 0.5 }
			OR = {
				has_equipment = { fighter_equipment > 1250 }
				has_deployed_air_force_size = { size > 1250 type = fighter }
			}
		}

		#max amount it can build
		NOT = {
			has_equipment = { strat_bomber_equipment > 5000 }
			has_deployed_air_force_size = { size > 5000 type = strategic_bomber }
		}
	}
	abort = {
		OR = {
			#conditions for building this
			AND = {
				has_country_flag = str_air_focus_1
				NOT = { num_of_military_factories > 60 }
			}
			AND = {
				NOT = { has_country_flag = str_air_focus_1 }
				OR = {
					NOT = { num_of_military_factories > 249 }
					NOT = { alliance_strength_ratio > 1.0 NOT = { any_neighbor_country = { has_war_with = ROOT } } }
				}
			}

			#prioritise building fighters
			AND = {
				enemies_strength_ratio > 0.5
				NOT = {
					has_equipment = { fighter_equipment > 1000 }
					has_deployed_air_force_size = { size > 1000 type = fighter }
				}
			}

			#max amount it can build
			has_equipment = { strat_bomber_equipment > 6000 }
			has_deployed_air_force_size = { size > 6000 type = strategic_bomber }
		}
	}

	ai_strategy = { type = equipment_production_factor      id = strategic_bomber 	         value = 101 }
	ai_strategy = { type = unit_ratio                       id = strategic_bomber 	         value = 200 }
	ai_strategy = { type = build_army                       id = strategic_bomber 	         value = 10000 }
}
