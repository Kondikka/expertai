	ai_strategy = { type = unit_ratio                       id = carrier 			         value = -200 }
	ai_strategy = { type = unit_ratio                       id = capital_ship 		         value = -200 }
	ai_strategy = { type = unit_ratio                       id = screen_ship 		         value = -200 }
	ai_strategy = { type = unit_ratio                       id = submarine 			         value = -200 }
}

carrier_ship_production = {
	enable = {
		has_tech = basic_carrier
		is_major = yes

		OR = {
			has_country_flag = carrier_navy_focus_1
			num_of_naval_factories > 40
		}
	}
	abort = {
		OR = {
			NOT = { is_major = yes }

			AND = {
				NOT = { has_country_flag = carrier_navy_focus_1 }
				NOT = { num_of_naval_factories > 40 }
			}
		}
	}

	ai_strategy = { type = equipment_production_factor      id = carrier 			         value = 0 }
	ai_strategy = { type = unit_ratio                       id = carrier 			         value = 200 }
}

capital_ship_production = {
	enable = {
		has_tech = basic_battleship
		is_major = yes

		OR = {
			has_country_flag = capital_navy_focus_1
			num_of_naval_factories > 30
		}
	}
	abort = {
		OR = {
			NOT = { is_major = yes }

			AND = {
				NOT = { has_country_flag = capital_navy_focus_1 }
				NOT = { num_of_naval_factories > 30 }
			}
		}
	}

	ai_strategy = { type = equipment_production_factor      id = capital_ship 		         value = 0 }
	ai_strategy = { type = unit_ratio                       id = capital_ship 		         value = 200 }
}

screen_ship_production = {
	enable = { always = yes }

	ai_strategy = { type = equipment_production_factor      id = screen_ship 		         value = 0 }
	ai_strategy = { type = unit_ratio                       id = screen_ship 		         value = 200 }
}

misc_ship_production = {
	enable = { always = no }

	ai_strategy = { type = equipment_production_factor      id = submarine 			         value = 0 }
	ai_strategy = { type = unit_ratio                       id = submarine 			         value = 200 }
}
