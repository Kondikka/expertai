### FIGHTERS
EAI_FTR_aircraft_production_strategy = {
	enable = {
		EAI_FTR_aircraft_production = yes
		
		NOT = { has_tech = jet_fighter1 }
		
		has_deployed_air_force_size = { size < 15000 type = fighter }
	}
	abort = {
		NOT = { 
			AND = {
				EAI_FTR_aircraft_production = yes
				
				NOT = { has_tech = jet_fighter1 }
				
				has_deployed_air_force_size = { size < 17500 type = fighter }
			}
		}
	}

	ai_strategy = { type = equipment_variant_production_factor id = fighter_equipment            value = 100 }
}

EAI_TAC_aircraft_production_strategy = {
	enable = {
		EAI_TAC_aircraft_production = yes
		
		NOT = { has_tech = jet_tactical_bomber1 }
		
		has_deployed_air_force_size = { size < 3000 type = tactical_bomber }
		
		if = { limit = { EAI_FTR_aircraft_production = yes enemies_strength_ratio > 0.5 }
			has_deployed_air_force_size = { size > 1250 type = fighter }
		}
	}
	abort = {
		NOT = {
			AND = {
				EAI_TAC_aircraft_production = yes
				
				NOT = { has_tech = jet_tactical_bomber1 }
		
				has_deployed_air_force_size = { size < 4500 type = tactical_bomber }
				
				if = { limit = { EAI_FTR_aircraft_production = yes enemies_strength_ratio > 0.5 }
					has_deployed_air_force_size = { size > 1000 type = fighter }
				}
			}
		}
	}

	ai_strategy = { type = equipment_variant_production_factor id = tac_bomber_equipment         value = 4 }
}
