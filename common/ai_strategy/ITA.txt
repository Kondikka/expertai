			ITA = { has_capitulated = yes }

			GER = { exists = no }
			GER = { is_subject = yes }
		}
	}
	abort = {
		NOT = {
			AND = {
				GER = { has_government = fascism }
				ITA = { has_government = fascism }
				
				NOT = { 
					ITA = { has_capitulated = yes }
					ITA = { exists = no }
					ITA = { is_subject = yes }
					
					GER = { has_capitulated = yes }
					GER = { exists = no }
					GER = { is_subject = yes }
				}
			}
		}
	}
	
	ai_strategy = { type = alliance id = "GER" value = -500 }
	
	ai_strategy = { type = alliance id = "ITA" value = -500 }
}
