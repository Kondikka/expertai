on_actions = {
	on_startup = {
		effect = {
			#EAI news event
			every_country = { limit = { is_ai = no } country_event = EAI_M.0 }
			random_country = { limit = { is_ai = no } set_country_flag = customisation_event }

			#add resistance reduction for AI
			every_country  = { limit = { is_ai = yes NOT = { tag = NOR tag = SWI tag = POL tag = CHI tag = FIN tag = GRE tag = SPR } } add_ideas = ns }
			every_country  = { limit = { is_ai = yes OR = { tag = NOR tag = SWI tag = POL tag = CHI tag = FIN tag = GRE tag = SPR } } add_ideas = ns2 } #highly defensive countries

			#AI neutrality ideas
			USA = { country_event = EAI_DM.0 }

			#AI starting construction queue
			GER = { country_event = EAI_C.0 }

			#default EAI settings
			set_global_flag = EAI_lend_lease
			#set_global_flag = EAI_free_convoys #hopefully not necessary anymore

			#radar script
			every_state = {
				limit = { radar_location = yes }
				set_state_flag = radar_site
			}

			#save the original location of a country for the mobilization script
			every_country = { limit = { capital_scope = { is_on_continent = europe } } set_country_flag = origin_europe }
			every_country = { limit = { capital_scope = { is_on_continent = north_america } } set_country_flag = origin_north_america }
			every_country = { limit = { capital_scope = { is_on_continent = south_america } } set_country_flag = origin_south_america }
			every_country = { limit = { capital_scope = { is_on_continent = africa } } set_country_flag = origin_africa }
			every_country = { limit = { capital_scope = { is_on_continent = asia } } set_country_flag = origin_asia }
			every_country = { limit = { capital_scope = { is_on_continent = middle_east } } set_country_flag = origin_middle_east }
		}
	}
	on_puppet = {
		effect = { #fix JAP-CHI peace conference
			if = {
				limit = {
					FROM = { tag = JAP is_ai = yes }
					ROOT = { OR = { tag = GXC tag = YUN tag = PRC tag = XSM } is_ai = yes }
				}

				CHI = { annex_country = { target = ROOT } }
			}
			if = {
				limit = {
					FROM = { tag = JAP is_ai = yes }
					ROOT = { tag = CHI is_ai = yes }
				}

				#resource rich states
				if = { limit = { 599 = { CONTROLLER = { OR = { tag = JAP is_subject_of = JAP } } } }
					599 = { JAP = { transfer_state = PREV } }
				}
				if = { limit = { 600 = { CONTROLLER = { OR = { tag = JAP is_subject_of = JAP } } } }
					600 = { JAP = { transfer_state = PREV } }
				}
				if = { limit = { 606 = { CONTROLLER = { OR = { tag = JAP is_subject_of = JAP } } } }
					606 = { JAP = { transfer_state = PREV } }
				}
			}
		}
	}
	on_annex = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = JAP is_ai = yes }
					FROM = { tag = PRC is_ai = yes }
				}

				#PRC territories
				if = { limit = { 622 = { CONTROLLER = { OR = { tag = JAP is_subject_of = JAP } } } }
					622 = { CHI = { transfer_state = PREV } }
				}
				if = { limit = { 615 = { CONTROLLER = { OR = { tag = JAP is_subject_of = JAP } } } }
					615 = { CHI = { transfer_state = PREV } }
				}
				if = { limit = { 621 = { CONTROLLER = { OR = { tag = JAP is_subject_of = JAP } } } }
					621 = { CHI = { transfer_state = PREV } }
				}
			}
		}
	}
	on_declare_war = {
		effect = {
			#AI allies should immediately join wars to avoid problems with their fronts
			if = { limit = { ROOT = { is_ai = yes } }
				ROOT = { country_event = { id = EAI_DM.5 hours = 1 } } #has to be delayed for it to work
			}
			if = { limit = { FROM = { is_ai = yes } }
				FROM = { country_event = { id = EAI_DM.5 hours = 1 } }
			}

			#invaded countries should instantly raise conscription law, waiting for the MTTH event takes too long
			if = { limit = { FROM = { is_ai = yes } }
				FROM = {
					country_event = { id = EAI_L.5 hours = 1 }
					country_event = { id = EAI_L.6 hours = 2 }
					country_event = { id = EAI_L.7 hours = 3 }
				}
			}

			#ATTEMPTS TO JOIN FACTION ON ILLEGAL WARDEC ON SELF
			if = {
				limit = {
					NOT = { any_state = { is_owned_by = FROM OR = { is_core_of = ROOT is_claimed_by = ROOT } } } #illegal

					NOT = { FROM = { is_in_faction = yes } }

					OR = {
						FROM = { major_country = yes }
						threat > 0.9
					}
				}

				FROM = {
					if = {
						limit = { has_government = democratic }
						set_country_flag = EAI_seeks_alliance_democratic
					}
					if = {
						limit = { has_government = communism }
						set_country_flag = EAI_seeks_alliance_communist
					}
					if = {
						limit = { has_government = fascism }
						set_country_flag = EAI_seeks_alliance_communist
					}
					if = {
						limit = { has_government = neutrality }
						set_country_flag = EAI_seeks_alliance_democratic
						set_country_flag = EAI_seeks_alliance_communist
						set_country_flag = EAI_seeks_alliance_fascist
					}

					country_event = EAI_DM.10101
				}
			}

			#ATTEMPTS TO JOIN FACTION ON ILLEGAL WARDEC ON NEARBY COUNTRY
			if = {
				limit = {
					NOT = { any_state = { is_owned_by = FROM OR = { is_core_of = ROOT is_claimed_by = ROOT } } } #illegal

					FROM = { has_country_flag = EAI_illegal_wargoal_justified_by@ROOT } #national focuses give wargoals but they should be prevented from triggering this

					threat > 0.9

					FROM = { strength_ratio = { tag = ROOT ratio < 0.5 } } #target is weak compared to invader

					ROOT = { has_added_tension_amount > 25 } #invader has already added a lot of threat

					any_country = {
						OR = {
							AND = { democratic_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_democratic } }
							AND = { communism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_communist } }
							AND = { fascism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_fascist } }
							AND = { #which faction a neutral seeks depending on the invader's ideology
								neutrality_willing_to_join_faction = yes
								if = { limit = { ROOT = { has_government = democratic } }
									NOT = { has_country_flag = EAI_seeks_alliance_fascist }
									NOT = { has_country_flag = EAI_seeks_alliance_commuist }
								}
								if = { limit = { ROOT = { has_government = communism } }
									NOT = { has_country_flag = EAI_seeks_alliance_fascist }
									NOT = { has_country_flag = EAI_seeks_alliance_democratic }
								}
								if = { limit = { ROOT = { has_government = fascism } }
									NOT = { has_country_flag = EAI_seeks_alliance_democratic }
									NOT = { has_country_flag = EAI_seeks_alliance_commuist }
								}
								if = { limit = { ROOT = { has_government = neutrality } }
									NOT = { has_country_flag = EAI_seeks_alliance_democratic }
									NOT = { has_country_flag = EAI_seeks_alliance_fascist }
									NOT = { has_country_flag = EAI_seeks_alliance_commuist }
								}
							}
						}
					}
				}

				FROM = { save_global_event_target_as = EAI_illegal_wardec_target }

				every_country = {
					limit = { democratic_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_democratic } }

					set_country_flag = EAI_seeks_alliance_democratic

					clr_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = { id = EAI_DM.10100 hours = 1 }
				}

				every_country = {
					limit = { communism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_communist } }

					set_country_flag = EAI_seeks_alliance_communist

					clr_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = { id = EAI_DM.10100 hours = 1 }
				}

				every_country = {
					limit = { fascism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_fascist } }

					set_country_flag = EAI_seeks_alliance_fascist

					clr_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = { id = EAI_DM.10100 hours = 1 }
				}

				every_country = {
					limit = {
						neutrality_willing_to_join_faction = yes
						if = { limit = { ROOT = { has_government = democratic } }
							NOT = { has_country_flag = EAI_seeks_alliance_fascist }
							NOT = { has_country_flag = EAI_seeks_alliance_commuist }
						}
						if = { limit = { ROOT = { has_government = communism } }
							NOT = { has_country_flag = EAI_seeks_alliance_fascist }
							NOT = { has_country_flag = EAI_seeks_alliance_democratic }
						}
						if = { limit = { ROOT = { has_government = fascism } }
							NOT = { has_country_flag = EAI_seeks_alliance_democratic }
							NOT = { has_country_flag = EAI_seeks_alliance_commuist }
						}
						if = { limit = { ROOT = { has_government = neutrality } }
							NOT = { has_country_flag = EAI_seeks_alliance_democratic }
							NOT = { has_country_flag = EAI_seeks_alliance_fascist }
							NOT = { has_country_flag = EAI_seeks_alliance_commuist }
						}
					}

					if = { limit = { ROOT = { has_government = democratic } }
						set_country_flag = EAI_seeks_alliance_fascist
						set_country_flag = EAI_seeks_alliance_commuist
					}
					if = { limit = { ROOT = { has_government = communism } }
						set_country_flag = EAI_seeks_alliance_fascist
						set_country_flag = EAI_seeks_alliance_democratic
					}
					if = { limit = { ROOT = { has_government = fascism } }
						set_country_flag = EAI_seeks_alliance_democratic
						set_country_flag = EAI_seeks_alliance_commuist
					}
					if = { limit = { ROOT = { has_government = neutrality } }
						set_country_flag = EAI_seeks_alliance_democratic
						set_country_flag = EAI_seeks_alliance_fascist
						set_country_flag = EAI_seeks_alliance_commuist
					}

					clr_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = { id = EAI_DM.10100 hours = 1 }
				}
			}
		}
	}
	on_justifying_wargoal_pulse = {
		effect = {
			if = { #add infamy
				limit = {
					NOT = { any_state = { is_owned_by = FROM OR = { is_core_of = ROOT is_claimed_by = ROOT } } } #illegal

					NOT = { FROM = { has_country_flag = EAI_infamy_added@ROOT } }
				}

				FROM = { set_country_flag = EAI_infamy_added@ROOT }

				if = { #add to infamy count
					limit = { NOT = { has_country_flag = EAI_infamy_count } }

					set_country_flag = EAI_infamy_count

					else = { modify_country_flag = { flag = EAI_infamy_count value = 1 } }
				}

				if = { #to make sure the script below doesnt fire below infamy limit
					limit = { is_major = no FROM = { is_major = no } has_country_flag = { flag = EAI_infamy_count value < 2 } }

					FROM = { set_country_flag = EAI_illegal_wargoal_target_country_by@ROOT }
				}
			}

			if = { #WILLING TO GUARANTEE TARGET ON ILLEGAL JUSTIFICATION: WARNING
				limit = {
					OR = { #dont fire this if the country is willing to join a faction instead
						NOT = { threat > 0.9 }

						NOT = { FROM = { strength_ratio = { tag = ROOT ratio < 0.5 } } }

						NOT = { ROOT = { has_added_tension_amount > 25 } }

						NOT = {
							any_country = {
								OR = {
									AND = { democratic_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_democratic } }
									AND = { communism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_communist } }
									AND = { fascism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_seeks_alliance_fascist } }
									AND = { #which faction a neutral seeks depending on the invader's ideology
										neutrality_willing_to_join_faction = yes
										if = { limit = { ROOT = { has_government = democratic } }
											NOT = { has_country_flag = EAI_seeks_alliance_fascist }
											NOT = { has_country_flag = EAI_seeks_alliance_commuist }
										}
										if = { limit = { ROOT = { has_government = communism } }
											NOT = { has_country_flag = EAI_seeks_alliance_fascist }
											NOT = { has_country_flag = EAI_seeks_alliance_democratic }
										}
										if = { limit = { ROOT = { has_government = fascism } }
											NOT = { has_country_flag = EAI_seeks_alliance_democratic }
											NOT = { has_country_flag = EAI_seeks_alliance_commuist }
										}
										if = { limit = { ROOT = { has_government = neutrality } }
											NOT = { has_country_flag = EAI_seeks_alliance_democratic }
											NOT = { has_country_flag = EAI_seeks_alliance_fascist }
											NOT = { has_country_flag = EAI_seeks_alliance_commuist }
										}
									}
								}
							}
						}
					}

					#requirements for this
					NOT = { any_state = { is_owned_by = FROM OR = { is_core_of = ROOT is_claimed_by = ROOT } } } #illegal

					NOT = { has_global_flag = EAI_illegal_wargoal_event_active } #fire this event chain only one at a time

					NOT = { FROM = { has_country_flag = EAI_illegal_wargoal_target_country_by@ROOT } } #fire only once per invader against this country

					any_country = { any_willing_to_guarantee_nation = yes }

					#rules
					OR = {
						ROOT = { is_major = yes }

						AND = {
							ROOT = { is_major = no }

							OR = {
								FROM = { is_major = yes }

								AND = {
									FROM = { is_major = no }
									has_country_flag = { flag = EAI_infamy_count value > 1 }
								}
							}
						}
					}
				}

				set_global_flag = EAI_illegal_wargoal_event_active

				FROM = { set_country_flag = EAI_illegal_wargoal_target_country_by@ROOT }

				FROM = { save_global_event_target_as = EAI_illegal_wargoal_target }

				ROOT = { save_global_event_target_as = EAI_illegal_wargoal_invader }

				if = { #create a "leader" from a random defender for event description
					limit = { any_country = { is_major = yes any_willing_to_guarantee_nation = yes } }

					random_country = {
						limit = { is_major = yes any_willing_to_guarantee_nation = yes }

						save_global_event_target_as = EAI_illegal_wargoal_defender
					}

					else = { #else choose a minor
						random_country = {
							limit = { any_willing_to_guarantee_nation = yes }

							save_global_event_target_as = EAI_illegal_wargoal_defender
						}
					}
				}

				country_event = EAI_DM.10000 #warning to invader

				every_other_country = { #global news event
					country_event = EAI_DM.10010
				}
			}

			if = { #WILLING TO JOIN FACTION ON ILLEGAL WARDEC: WARNING
				limit = {
					NOT = { any_state = { is_owned_by = FROM OR = { is_core_of = ROOT is_claimed_by = ROOT } } } #illegal

					threat > 0.9

					FROM = { strength_ratio = { tag = ROOT ratio < 0.5 } } #target is weak compared to invader

					ROOT = { has_added_tension_amount > 25 } #invader has already added a lot of threat

					any_country = {
						NOT = { has_country_flag = EAI_faction_join_warning_issued@ROOT }
						OR = {
							democratic_willing_to_join_faction = yes
							communism_willing_to_join_faction = yes
							fascism_willing_to_join_faction = yes
							neutrality_willing_to_join_faction = yes
						}
					}
				}

				FROM = { country_event = { id = EAI_DM.10008 days = 20 } } #check if still justifying after 20 days

				every_country = {
					limit = { democratic_willing_to_join_faction = yes NOT = { has_country_flag = EAI_faction_join_warning_issued@ROOT } }

					set_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = EAI_DM.10007
				}

				every_country = {
					limit = { communism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_faction_join_warning_issued@ROOT } }

					set_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = EAI_DM.10007
				}

				every_country = {
					limit = { fascism_willing_to_join_faction = yes NOT = { has_country_flag = EAI_faction_join_warning_issued@ROOT } }

					set_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = EAI_DM.10007
				}

				every_country = {
					limit = { neutrality_willing_to_join_faction = yes NOT = { has_country_flag = EAI_faction_join_warning_issued@ROOT } }

					set_country_flag = EAI_faction_join_warning_issued@ROOT

					country_event = EAI_DM.10007
				}
			}

			if = { #flag a country when illegally justified against so the on_declare_war script knows when a wargoal was illegally created
				limit = {
					NOT = { any_state = { is_owned_by = FROM OR = { is_core_of = ROOT is_claimed_by = ROOT } } } #illegal

					threat > 0.9

					FROM = { strength_ratio = { tag = ROOT ratio < 0.5 } } #target is weak compared to invader

					ROOT = { has_added_tension_amount > 25 } #invader has already added a lot of threat

					NOT = { FROM = { has_country_flag = EAI_illegal_wargoal_justified_by@ROOT } }
				}

				FROM = { set_country_flag = EAI_illegal_wargoal_justified_by@ROOT }
				FROM = { country_event = { id = EAI_DM.10008 days = 20 } #if justification was cancelled, clear flag
			}
		}
	}
}
