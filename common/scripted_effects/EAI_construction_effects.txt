EAI_queue_AIR = {
	if = {
		limit = {
			any_state = {
				EAI_available_AIR = yes
				
				air_base > 0
				
				any_neighbor_state = { CONTROLLER = { has_war_with = ROOT } }
			}
		}
		
		random_state = {
			limit = {
				EAI_available_AIR = yes
				
				air_base > 0
				
				any_neighbor_state = { CONTROLLER = { has_war_with = ROOT } }
			}
			
			EAI_add_AIR = yes
		}
	}
	
	else_if = {
		limit = {
			any_state = {
				EAI_available_AIR = yes
				
				any_neighbor_state = { CONTROLLER = { has_war_with = ROOT } }
			}
		}
	
		random_state = {
			limit = {
				EAI_available_AIR = yes
				
				any_neighbor_state = { CONTROLLER = { has_war_with = ROOT } }
			}
			
			EAI_add_AIR = yes
		}
	}
	
	else_if = {
		limit = {
			any_state = {
				EAI_available_AIR = yes
				
				any_state = { CONTROLLER = { has_war_with = ROOT } distance_to = { target = PREV value < 500 } }
			}
		}
	
		random_state = {
			limit = {
				EAI_available_AIR = yes
				
				any_state = { CONTROLLER = { has_war_with = ROOT } distance_to = { target = PREV value < 500 } }
			}
			
			EAI_add_AIR = yes
		}
	}
	
	else_if = {
		limit = {
			any_state = {
				EAI_available_AIR = yes
				
				any_state = { 
					CONTROLLER = { 
						OR = {
							AND = { EAI_communist_nation = yes ROOT = { NOT = { EAI_communist_nation = yes } } }
							AND = { EAI_fascist_nation = yes ROOT = { NOT = { EAI_fascist_nation = yes } } }
							AND = { EAI_democratic_nation = yes ROOT = { NOT = { EAI_democratic_nation = yes } } }
						}
						
						has_deployed_air_force_size = { size > 1000 }
					} 
					
					distance_to = { target = PREV value < 500 }
				}
			}
		}
	
		random_state = {
			limit = {
				EAI_available_AIR = yes
				
				any_state = { 
					CONTROLLER = { 
						OR = {
							AND = { EAI_communist_nation = yes ROOT = { NOT = { EAI_communist_nation = yes } } }
							AND = { EAI_fascist_nation = yes ROOT = { NOT = { EAI_fascist_nation = yes } } }
							AND = { EAI_democratic_nation = yes ROOT = { NOT = { EAI_democratic_nation = yes } } }
						}
						
						has_deployed_air_force_size = { size > 1000 }
					} 
					
					distance_to = { target = PREV value < 500 }
				}
			}
			
			EAI_add_AIR = yes
		}
	}
}

EAI_queue_INF = { # queue INF anywhere possible
	if = {
		limit = { any_state = { EAI_available_INF = yes } }
		if = {
			limit = { any_state = { EAI_available_INF = yes EAI_available_state_0 = yes is_in_home_area = yes } }
			random_state = { limit = { EAI_available_INF = yes EAI_available_state_0 = yes is_in_home_area = yes } EAI_add_INF = yes }
		}
		else_if = {
			limit = { any_state = { EAI_available_INF = yes is_in_home_area = yes } }
			random_state = { limit = { EAI_available_INF = yes is_in_home_area = yes } EAI_add_INF = yes }
		}
		else_if = {
			limit = { any_state = { EAI_available_INF = yes EAI_available_state_0 = yes } }
			random_state = { limit = { EAI_available_INF = yes EAI_available_state_0 = yes } EAI_add_INF = yes }
			
			else = { random_state = { limit = { EAI_available_INF = yes } EAI_add_INF = yes } }
		}
	}
}

EAI_queue_INF_2 = { # see if there is a good place for infrastructure, else queue factories
	if = {
		limit = {
			ROOT = { num_of_civilian_factories > 9 }
			EAI_available_INF = yes
			is_in_home_area = yes
		}

		EAI_add_INF = yes
	}
	else_if = {
		limit = { EAI_available_state_0 = yes ROOT = { num_of_civilian_factories < 10 } }

		EAI_add_CIC = yes
	}
	else_if = {
		limit = { EAI_available_state_0 = yes }

		EAI_add_MIC = yes
	}
	else = {
		random_state = {
			limit = {
				EAI_available_INF = yes
				is_in_home_area = yes
			}
			EAI_add_INF = yes
		}
	}
}
