queue_INF = { # queue INF anywhere possible
	if = {
		limit = { any_state = { available_INF = yes } }
		if = {
			limit = { any_state = { available_INF = yes available_state_0 = yes is_in_home_area = yes } } random_state = { limit = { available_INF = yes available_state_0 = yes is_in_home_area = yes } add_INF = yes }
			else = {
				if = {
					limit = { any_state = { available_INF = yes is_in_home_area = yes } } random_state = { limit = { available_INF = yes is_in_home_area = yes } add_INF = yes }
					else = {
						if = {
							limit = { any_state = { available_INF = yes available_state_0 = yes } } random_state = { limit = { available_INF = yes available_state_0 = yes } add_INF = yes }
							else = {
								random_state = { limit = { available_INF = yes } add_INF = yes }
							}
						}
					}
				}
			}
		}
		else = {
			queue_RADAR = yes
		}
	}
}

queue_INF_2 = { # see if there is a good place for infrastructure, else queue factories
	if = {
		limit = {
			ROOT = { num_of_civilian_factories > 9 }
			available_INF = yes
			is_in_home_area = yes
		}

		add_INF = yes

		else = {
			if = {
				limit = { available_state_0 = yes ROOT = { num_of_civilian_factories < 10 } }

				add_CIC = yes

				else = {
					if = {
						limit = { available_state_0 = yes }

						add_MIC = yes

						else = {
							random_state = {
								limit = {
									available_INF = yes
									is_in_home_area = yes
								}
								add_INF = yes
							}
						}
					}
				}
			}
		}
	}
}

queue_INF_supply = { # for building INF to increase supply
	if = {
		limit = {
			any_state = {
				available_INF = yes
				free_building_slots = { building = infrastructure size > 5 include_locked = no }
				ROOT = { divisions_in_state = { state = PREV size > 5 } }
				any_neighbor_state = { CONTROLLER = { has_war_with = ROOT strength_ratio = { tag = ROOT ratio > 0.5 } } }
			}
		}
		
		random_state = {
			limit = {
				available_INF = yes
				free_building_slots = { building = infrastructure size > 5 include_locked = no }
				ROOT = { divisions_in_state = { state = PREV size > 5 } }
				any_neighbor_state = { CONTROLLER = { has_war_with = ROOT strength_ratio = { tag = ROOT ratio > 0.5 } } }
			}
			add_INF = yes
		}
		
		else = {
			if = {
				limit = {
					any_state = {
						available_INF = yes
						free_building_slots = { building = infrastructure size > 5 include_locked = no }
						ROOT = { divisions_in_state = { state = PREV size > 5 } }
						any_neighbor_state = { CONTROLLER = { has_war_with = ROOT } }
					}
				}
				random_state = {
					limit = {
						available_INF = yes
						free_building_slots = { building = infrastructure size > 5 include_locked = no }
						ROOT = { divisions_in_state = { state = PREV size > 5 } }
						any_neighbor_state = { CONTROLLER = { has_war_with = ROOT } }
					}
					add_INF = yes
				}
			}
		}
	}
}
