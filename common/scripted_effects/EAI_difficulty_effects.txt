# two equipment cost reduction ideas cant be allowed to be active at the same time so the highest is used
EAI_fix_production_boost = { 
	if = { 
		limit = { 
			has_country_flag = EAI_difficulty_bonus_hard_active
			check_variable = { EAI_desperate_defense_level > 3 }
		}
		
		EAI_clear_equipment_modifier_HARD = yes
	}
	
	if = { 
		limit = { 
			has_country_flag = EAI_difficulty_bonus_very_hard_active
			check_variable = { EAI_desperate_defense_level > 6 }
		}
		
		EAI_clear_equipment_modifier_VERY_HARD = yes
	}
	
	if = { 
		limit = { 
			has_country_flag = EAI_difficulty_bonus_insane_active
			check_variable = { EAI_desperate_defense_level > 13 }
		}
		
		EAI_clear_equipment_modifier_INSANE = yes
	}
	
	##########################################################
	
	if = { 
		limit = { 
			check_variable = { EAI_desperate_defense_level < 4 }
			has_country_flag = EAI_difficulty_bonus_hard_active
		}
		
		EAI_clear_equipment_modifier_DD = yes
	}
	
	if = { 
		limit = { 
			check_variable = { EAI_desperate_defense_level < 7 }
			has_country_flag = EAI_difficulty_bonus_very_hard_active
		}
		
		EAI_clear_equipment_modifier_DD = yes
	}
	
	if = { 
		limit = { 
			check_variable = { EAI_desperate_defense_level < 14 }
			has_country_flag = EAI_difficulty_bonus_insane_active
		}
		
		EAI_clear_equipment_modifier_DD = yes
	}
}

EAI_get_custom_bonus_settings = {
	set_temp_variable = { settings = 25 }
	while_loop_effect = { limit = { check_variable = { settings > 0 } }
		meta_effect = {
			text = { var:EAI_configuration_target = { set_variable = { Root.EAI_custom_bonus_[x] = This.EAI_custom_bonus_[x] } } }
			x = "[?settings]"
		}
		subtract_from_temp_variable = { settings = 1 }
	}
}
EAI_set_custom_bonus_settings = {
	set_temp_variable = { settings = 25 }
	while_loop_effect = { limit = { check_variable = { settings > 0 } }
		meta_effect = {
			text = {

				ROOT = { clear_variable = EAI_custom_bonus_[x] }

			x = "[?settings]"
		}

	}
	var:EAI_configuration_target = { force_update_dynamic_modifier = yes }
}
